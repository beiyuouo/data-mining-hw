def pca_(dataset, k=2, svd=False):
    """
    nk
    :param dataset: n*m,n,m,np.array
    :param svd: 
    :return: k*mt
    """
    dataset = dataset.T
    dataset = pp.scale(dataset, axis=1)  # 
    cov_mat = np.cov(dataset)  # 
    if svd:  # SVD
        u, s, vt = np.linalg.svd(cov_mat)  # su
        eig_vals = s[: k]  # K
        red_eig_vects = u[:, : k]  #  n*k
        t = np.sum(eig_vals) / np.sum(s)  # 

    else:  # 
        eig_vals, eig_vects = np.linalg.eig(cov_mat)  # 
        eig_vals_ind = np.argsort(eig_vals)[:: -1][: k]  # k
        red_eig_vects = eig_vects[:, eig_vals_ind]  # 
        # PCA*
        t = np.sum(eig_vals[eig_vals_ind]) / np.sum(eig_vals)
    low_dim_data = np.dot(red_eig_vects.T, dataset)  # *=
    return low_dim_data.T, t
