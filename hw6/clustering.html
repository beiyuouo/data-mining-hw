estimator = KMeans(n_clusters=2) # 
estimator.fit(X)  # 
label_pred = estimator.labels_  # 
print(f'SSE: {estimator.inertia_}')
# k-means
x0 = X[label_pred == 0]
x1 = X[label_pred == 1]
plt.scatter(x0[:, 0], x0[:, 1], c="red", marker='o', label='class 1')  
plt.scatter(x1[:, 0], x1[:, 1], c="green", marker='o', label='class 2')
plt.xlabel('sepal length')  
plt.ylabel('sepal width')  
plt.legend(loc=2)
plt.show()
